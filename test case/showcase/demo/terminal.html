$('#trmnl').puiterminal({
    welcomeMessage: 'Welcome to PrimeUI Terminal, how are you today?',
    handler: function(request, response) {
        if($.trim(request).length) {
           if(request == 'date') {
               response.call(this, new Date());
           }
           else if (request.indexOf('greet') === 0) {
               response.call(this, 'Hello ' + request.split(' ')[1]);
           }
           else {
               response.call(this, "Unknown command: " + request);
           }
        }
    }
});

Usage

Terminal requires a handler to process the commands. 
<p-terminal handler="TerminalDemo.handler" welcomemessage="Welcome to Terminal Element"></terminal>
                                
var TerminalDemo = {
  
    handler: function(request, response) {
        if($.trim(request).length) {
            $.ajax({
                 type: "GET",
                 url: 'rest/terminal/' + request,
                 dataType: "text",
                 context: this,
                 success: function (data) {
                     response.call(this, data);
                 }
             }); 
         }
    }
    
};
