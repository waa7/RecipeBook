$('#btn-multiple').puibutton().click(function() {
    addMessage('info', [{summary: 'Message Title', detail: 'Message Detail here.'}

        ,{summary: 'Message Title', detail: 'Message Detail here.'}]);
});

addMessage = function(severity, msg) {
    $('#default').puimessages('show', severity, msg);
    document.getElementById('msgs').show(severity, msg);
}

Usage

Messages are displayed using show() method that takes severity and an array of messages as parameters. 
<p-messages id="msgs"></p-messages>
                                
//single
document.getElementById('msgs').show('info', {summary: 'Message Title', detail: 'Message Detail here.'});

//multiple
document.getElementById('msgs').show('info', [{summary: 'Message Title', detail: 'Message Detail here.'}
        ,{summary: 'Message Title', detail: 'Message Detail here.'}
        ,{summary: 'Message Title', detail: 'Message Detail here.'}]);
